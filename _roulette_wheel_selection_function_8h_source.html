173  Object generate() override
174  {
175  Score sum_score = 0;
176  for (auto& valued_object : valued_objects_) {
177 
178  if (valued_object.score_ < std::numeric_limits<Score>::epsilon()) {
179  valued_object.score_ = std::numeric_limits<Score>::epsilon();
180  }
181  sum_score += valued_object.score_;
182  }
183 
184  // if sum of score is zero, randomly select
185  if (std::abs(sum_score) < std::numeric_limits<Score>::epsilon()) {
186 
187  std::uniform_int_distribution<> uniform(0, valued_objects_.size() -1);
188  unsigned int index = uniform(generator_);
189  return valued_objects_.at(index).object_.object;
190  }
191  else {
192 
193  std::uniform_real_distribution<Score> uniform(0, sum_score);
194  Score rnd = uniform(generator_);
195 
196  Score cumulative_weight = 0;
197  for (auto& valued_object : valued_objects_) {
198 
199  cumulative_weight += valued_object.score_;
200  if (rnd < cumulative_weight ||
201  std::abs(rnd-cumulative_weight) < std::numeric_limits<Score>::epsilon()) {
202 
203  return valued_object.object_.object;
204  }
205  }
206  throw std::logic_error("RouletteWheelSelectionFunction generate error.");
207  }
